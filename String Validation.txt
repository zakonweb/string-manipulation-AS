DECLARE STR1 : STRING
DECLARE FLAG : BOOLEAN
DECLARE Count : INTEGER
DECLARE nextChar : CHARACTER

STR1 = ""
FLAG = TRUE
Count = 0
nextChar = ''

OUTPUT "Input a string in aaaa-999-AAA format: "
INPUT STR1

//LENGTH CHECK
IF LEN(STR1) <> 12
  THEN
    FLAG = FALSE
ENDIF

//FIXED CHARACTERS
IF Format = TRUE
  THEN
    IF MID(STR1,5,1) <> '-' OR MID(STR1,9,1) <> '-'
      THEN
        FLAG = FALSE
    ENDIF
ENDIF

//Check first group
IF Format = TRUE
  THEN
    FOR Count = 1 to 4
        nextChar = MID(STR1,Count,1)
        IF nextChar < 'a' OR nextChar > 'z'
          THEN
            FLAG = FALSE
        ENDIF
    END FOR
ENDIF

//Check second group
IF Format = TRUE
  THEN
    FOR Count = 6 to 8
        nextChar = MID(STR1,Count,1)
        IF nextChar < '0' OR nextChar > '9'
          THEN
            FLAG = FALSE
        ENDIF
    END FOR
ENDIF

//Check third group
IF Format = TRUE
  THEN
    FOR Count = 10 to 12
        nextChar = MID(STR1,Count,1)
        IF nextChar < 'A' OR nextChar > 'Z'
          THEN
            FLAG = FALSE
        ENDIF
    END FOR
ENDIF

IF FLAG = TRUE
  THEN
    OUTPUT "Valid String..."
  ELSE
    OUTPUT "Invalid String..."
ENDIF
